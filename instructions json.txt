nb : le client envoie une seule instruction à la fois au serveur, tandis que le serveur envoie une liste (array) d'instructions au client.

Transmissions client -> serveur

- connexion

	var connec = {
		type: 'connexion'
		pseudo: xxx,
		pswd: xxx,
	};

- message

	var msg = {
		type: 'message'
		txt: xxx,
	};

- envoyer signal "prêt"

	var ready = {
		type = 'readyToPlay'
	};

- vote

	var vote = {
		type = 'vote',
		pseudoVoted = xxx
	};


- quitter
	//inutile maintenant, les déconnexion sont gerées automatiquement


Transmissions serveur -> client

	var msg = {
		type: 'message'
		sender: xxx,
		date: xxx,
		txt: xxx,
		color: xxx,
	};


- activer/désactiver votes


	var va = {
		type: 'voteActivate',
		typeVote : xxx, //champs pour désigner si vote normal "normal", psychopathe "psycho" ou capacité spéciale "special"
		pseudoNominees: [ xxx, xxx, xxx ], //liste des condidats nominés
	};

	var vd = {
		type: 'voteDeactivate'
		typeVote : xxx //vote a desactiver
	};

- activer/désactiver "starter"

	var starterActivate = {
		type = 'starterActivate'
	};

	var starterDeactivate = {
		type = 'starterDeactivate'
	};


- MAJ profil perso

	var role = {
		type = 'role', //pour le rôle, cf fin du document
		role = xxx
	};

- MAJ liste joueurs

	var UserInfo {
		type = 'userInfo'
		pseudo = xxx,
		role = xxx, //role du joueur (null si inconnu)
		alive = xxx //booléen
	};

- activer timer

	var timerActivate = {
		type = 'timerActivate',
		time = xxx //temps en secondes
	};

- message d'erreur

	var errorMsg = {
		type = 'errorMessage',
		txt = xxx
	};
