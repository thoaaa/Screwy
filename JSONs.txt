		type: 'message',

		txt: xxx
	};

		type = 'readyToPlay',
		sender = xxx
	};

- vote

	var vote = {
		type = 'vote',
		sender = xxx,
		pseudoVoted = xxx
	};

TRANSMISSIONS WEBWORKER > WEBWORKER > THREAD PRINCIPAL

- message

	var msg = {
		receivers : [xxx,xxx,xxx], //liste des destinataires
		instruction : {
			type: 'message',
			sender : xxx,
			date: xxx,
			txt: xxx
		}
	};

- activer/désactiver votes

	var va = {
		receivers : [xxx,xxx,xxx], //liste des destinataires
			instruction : {
				type: 'voteActivate',
				typeVote : xxx, //champs pour désigner si vote normal "normal", psychopathe "psycho" ou capacité spéciale "special"
				pseudoNominees: [ xxx, xxx, xxx ] //liste des condidats nominés
		}
	};

	var vd = {
		receivers : [xxx,xxx,xxx] //liste des destinataires
			instruction : {
				type: 'voteDeactivate',
				typeVote : xxx //vote a desactiver
		}
	};

- activer/désactiver "starter"

	var starterActivate = {
		receivers : [xxx,xxx,xxx] //liste des destinataires
		instruction : {
			type = 'starterActivate'
		}
	};

	var starterDeactivate = {
		receivers : [xxx,xxx,xxx] //liste des destinataires
		instruction : {
			type = 'starterDeactivate'
		}
	};


- MAJ profil perso

	var role = {
		receivers : [xxx,xxx,xxx] //liste des destinataires
		instruction : {
			type = 'role', //pour le rôle, cf fin du document
			role = xxx
		}
	};

- MAJ liste joueurs

	var UserInfo = {
		receivers : [xxx,xxx,xxx] //liste des destinataires
		instruction : {
			type = 'userInfo'
			pseudo = xxx,
			role = xxx, //role du joueur (null si inconnu)
			alive = xxx //booléen
		}
	};

- activer timer
